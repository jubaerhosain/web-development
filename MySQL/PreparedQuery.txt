const mysql = require('mysql');

const connection = mysql.createConnection({
  host: 'localhost',
  user: 'root',
  password: 'password',
  database: 'mydatabase'
});

// Prepare a statement with placeholders
const sql = 'SELECT * FROM customers WHERE country = ? AND city = ?';

// Define an array of values to replace the placeholders
const values = ['USA', 'New York'];

// Execute the prepared statement
connection.query(sql, values, (error, results, fields) => {
  if (error) throw error;
  console.log(results);
});

// Close the database connection
connection.end();

Using prepared statements in MySQL with Node.js is an effective way to prevent SQL injection attacks, as it ensures that user input values are properly escaped and sanitized before being used in SQL queries.
